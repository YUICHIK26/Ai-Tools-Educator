function countUp(id,target,sfx){

  let v=0, step=target/70;
  const t=setInterval(()=>{ v+=step; if(v>=target){el.textContent=target+sfx;clearInterval(t);}else el.textContent=Math.floor(v)+sfx; },16);
}
function initCounts(ids){
  const targets={ct1:200,ct2:18,ct3:150,ab1:200,ab2:18,ab3:150};
  const sfx={ct1:'+',ct2:'+',ct3:'+',ab1:'+',ab2:'+',ab3:'+'};
  const obs=new IntersectionObserver(function(entries){
    entries.forEach(function(x){
      if(x.isIntersecting){
        ids.forEach(function(id){ countUp(id, targets[id]||99, sfx[id]||''); });
        obs.disconnect();
      }
    });
  },{threshold:0.2});
  const el=document.getElementById(ids[0]); if(el)obs.observe(el);
}

function mountDrag(el,rot,onStart){
  const inner=el.querySelector('.d-inner');
  let vx=0,vy=0,lx=0,ly=0,dragging=false,raf=0;
  el.addEventListener('mousemove',function(e){
    if(dragging)return;
    const r=inner.getBoundingClientRect();
    inner.style.setProperty('--sx',(e.clientX-r.left)+'px');
    inner.style.setProperty('--sy',(e.clientY-r.top)+'px');
    inner.style.transform='perspective(900px) rotateX('+((e.clientY-r.top-r.height/2)/9)+'deg) rotateY('+((r.width/2-(e.clientX-r.left))/9)+'deg) scale(1.04)';
  });
  el.addEventListener('mouseleave',function(){ if(!dragging) inner.style.transform='rotate('+rot+'deg)'; });
  const dn=function(e){
    dragging=true; onStart(); cancelAnimationFrame(raf);
    lx=e.touches?e.touches[0].clientX:e.clientX;
    ly=e.touches?e.touches[0].clientY:e.clientY;
    document.addEventListener(e.touches?'touchmove':'mousemove',mv,{passive:true});
    document.addEventListener(e.touches?'touchend':'mouseup',up);
  };
  const mv=function(e){
    if(!dragging)return;
    const cx=e.touches?e.touches[0].clientX:e.clientX,cy=e.touches?e.touches[0].clientY:e.clientY;
    vx=cx-lx;vy=cy-ly;
    el.style.left=(el.offsetLeft+vx)+'px'; el.style.top=(el.offsetTop+vy)+'px';
    lx=cx;ly=cy;
  };
  const up=function(){
    dragging=false; inner.style.transform='rotate('+rot+'deg)';
    document.removeEventListener('mousemove',mv); document.removeEventListener('mouseup',up);
    document.removeEventListener('touchmove',mv); document.removeEventListener('touchend',up);
    glide();
  };
  const glide=function(){ if(Math.abs(vx)<0.12&&Math.abs(vy)<0.12)return; vx*=0.94;vy*=0.94; el.style.left=(el.offsetLeft+vx)+'px'; el.style.top=(el.offsetTop+vy)+'px'; raf=requestAnimationFrame(glide); };
  el.addEventListener('mousedown',dn);
  el.addEventListener('touchstart',dn,{passive:true});
}

function initForge(){
  const pgForge  = document.getElementById('pg-forge');
  const container= document.getElementById('forge-3d');
  const streak   = document.getElementById('forge-streak');
  const header   = document.getElementById('forge-hdr');
  const fstage   = document.getElementById('forge-stage');
  const lerp     = function(a,b,t){ return a+(b-a)*t; };

  let rawF=0, curF=0, rawG=0, curG=0, rafId=null;

  function applyForge(p){
    container.style.transform = 'rotateX('+(35-p*35)+'deg) scale('+(0.84+p*0.16)+')';
    const sm = -150 + p*300;
    streak.style.transform = 'translate('+sm+'%,'+sm+'%)';
    header.style.opacity = Math.max(0, 1-p*1.7).toFixed(3);
    header.style.transform = 'translateY('+(-p*42)+'px)';
  }

  function applyGem(gp){
    const offsets=[0,0.06,0.12,0.18,0.24];
    [1,2,3,4,5].forEach(function(n,i){
      const p=document.getElementById('gp'+n); if(!p) return;
      const l=p.getTotalLength();
      const adj=Math.max(0,Math.min(1,(gp-offsets[i])*1.6));
      p.style.strokeDashoffset = (l*(1-adj)).toFixed(2);
    });
    const big=document.getElementById('gem-big'), sm=document.getElementById('gem-sm');
    const show = gp > 0.58;
    big.style.opacity = show?'1':'0';
    big.style.transform = show?'translateY(0)':'translateY(20px)';
    sm.style.opacity = show?'1':'0';
  }

  function tick(){
    curF = lerp(curF, rawF, 0.11);
    curG = lerp(curG, rawG, 0.09);
    applyForge(curF);
    applyGem(curG);
    const dF = Math.abs(curF-rawF), dG = Math.abs(curG-rawG);
    if(dF > 0.0004 || dG > 0.0004) rafId = requestAnimationFrame(tick);
    else rafId = null;
  }

  pgForge.addEventListener('scroll', function(){
    const st = pgForge.scrollTop;
    rawF = Math.min(st / (window.innerHeight*0.85), 1);
    const gstage  = document.getElementById('gem-stage');
    const goffset = fstage.offsetHeight + window.innerHeight*0.22;
    rawG = Math.max(0, Math.min(1, (st-goffset) / Math.max(1, gstage.offsetHeight-window.innerHeight)));
    if(!rafId) rafId = requestAnimationFrame(tick);
  }, {passive:true});
}

function buildToolGrid(gridId, filter){
  var grid=document.getElementById(gridId); if(!grid) return;
  var filtered=filter==='all'?TOOLS_DATA:TOOLS_DATA.filter(function(t){return t.cat===filter;});
  grid.innerHTML=filtered.map(function(t){
    var tutUrl=TUTS[t.name]||'';
    var tutBtn=tutUrl ? '<button class="tut-btn" data-name="'+encodeURIComponent(t.name)+'" data-url="'+encodeURIComponent(tutUrl)+'">&#9654; Tutorial</button>' : '';
    var tagCls=t.free?'free':'paid';
    var tagTxt=t.free?'Free tier':'Paid';
    return '<div class="tool-card rv"><div class="tool-header"><div class="tool-icon">'+t.icon+'</div><div><div class="tool-name">'+t.name+'</div><div class="tool-cat">'+t.cat+'</div></div></div><div class="tool-desc">'+t.desc+'</div><div class="tool-footer"><div class="tool-price">'+t.price+'</div><div class="tool-tags">'+tutBtn+'<div class="tool-tag '+tagCls+'">'+tagTxt+'</div></div></div></div>';
  }).join('');
  /* Wire tutorial buttons via event delegation to avoid inline quote issues */
  grid.querySelectorAll('.tut-btn').forEach(function(btn){
    btn.addEventListener('click',function(e){
      e.stopPropagation();
      openVid(decodeURIComponent(btn.dataset.name), decodeURIComponent(btn.dataset.url));
    });
  });
  setTimeout(function(){ grid.querySelectorAll('.rv').forEach(function(el){el.classList.add('show');}); },60);
}

   CHAT TOOLBAR FUNCTIONS
══════════════════════════════════════ */
var _voiceOn=false, _videoOn=false, _SR=null;

function toggleVoice(){
  _voiceOn=!_voiceOn;
  var btn=document.getElementById('voice-btn');
  if(!btn) return;
  btn.classList.toggle('voice-on',_voiceOn);
  if(_voiceOn){
    toast('🎙️','Voice input active');
    var SRC=window.SpeechRecognition||window.webkitSpeechRecognition;
    if(SRC){
      _SR=new SRC(); _SR.continuous=false; _SR.interimResults=false;
      _SR.onresult=function(e){
        var t=e.results[0][0].transcript;
        var inp=document.getElementById('chat-input');
        if(inp){ inp.value=t; chatResize(inp); }
        toggleVoice();
      };
      _SR.onerror=function(){ toggleVoice(); };
      _SR.onend=function(){ if(_voiceOn) toggleVoice(); };
      _SR.start();
    }
  } else {
    if(_SR){ _SR.stop(); _SR=null; }
    toast('🎙️','Voice stopped');
  }
}

function toggleVideo(){
  _videoOn=!_videoOn;
  var btn=document.getElementById('video-btn');
  var dot=document.getElementById('vid-dot');
  if(btn) btn.style.borderColor=_videoOn?'rgba(224,80,64,.45)':'';
  if(btn) btn.style.color=_videoOn?'#e05040':'';
  if(dot) dot.style.opacity=_videoOn?'1':'0';
  toast(_videoOn?'📷':'📷',_videoOn?'Video ON':'Video OFF');
}

function stopStreaming(){
  if(window._si){ clearInterval(window._si); window._si=null; }
  var sb=document.getElementById('stop-btn'); if(sb) sb.classList.remove('show');
  toast('⏹️','Response stopped');
}
